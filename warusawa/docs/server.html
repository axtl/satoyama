urls = ('/post-(d+)/comment-(d+)/?', 'comment', '/post-(d+)/comments/?', 'comments', '/post-(d+)/?', 'post', '/posts/?', 'posts', '/', 'index')

class index:

def GET(self): return stache.Index().render()

class posts:

def GET(self): return "posts? what posts?" return stache.Index

class post:

def GET(self, post_id): return "post_id: " + str(post_id)

class comments:

def GET(self, for_post_id): return "comments for_post_id: " + str(for_post_id)

class comment:

def GET(self, for_post_id, comm_id): return "for_post_id: " + str(for_post_id) + " comm_id: " + str(comm_id)
