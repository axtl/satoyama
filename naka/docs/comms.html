comms_delete = (pid) ->
    r.c.lrange r.post_key(pid, 'comm.list'), 0, -1, (err, comm_list) ->
        done = 0
        for cid in comm_list
            done++
            do (cid) ->
                comm_mod.comm_delete(pid, cid)
                if done == comm_list.length
                    r.c.del r.post_key(pid, 'comm.list')
